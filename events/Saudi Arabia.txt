	id = saudi.1

	trigger = {
		tag = SAU
		has_completed_focus = SAU_the_concession_question

		}

				add_opinion_modifier = { target = SAU modifier = SAU_concession_rights_to_japan }
		}

	}
	option = { # Give concession rights to Texas Company and Socal (USA)
		name = saudi.1.b
		SAU = {
			give_resource_rights = { receiver = USA state = 798 }
			give_resource_rights = { receiver = USA state = 292 }
		}
		USA = {
				add_opinion_modifier = { target = SAU modifier = SAU_concession_rights_to_american_companies }
		}
		SAU_oil_money_level_up = yes
	}
